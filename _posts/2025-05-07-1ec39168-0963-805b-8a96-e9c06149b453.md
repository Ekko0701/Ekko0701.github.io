from collections import deque

def bfs(graph, start):
    visited = set()           # 방문한 노드를 기록할 집합
    queue = deque([start])    # 시작 노드를 큐에 넣기

    while queue:
        node = queue.popleft()  # 큐에서 노드 꺼내기
        if node not in visited:
            print(node, end=' ')  # 방문 시 출력
            visited.add(node)
            queue.extend(neighbor for neighbor in graph[node] if neighbor not in visited)

# 그래프 예시 (인접 리스트)
graph = {
    'A': ['B', 'C'],
    'B': ['A', 'D', 'E'],
    'C': ['A', 'F'],
    'D': ['B'],
    'E': ['B', 'F'],
    'F': ['C', 'E']
}

bfs(graph, 'A')
